select SUM(etime)
from daily_activity_concede
inner join site_info
on daily_activity_concede.pid = site_info.p_id
inner join open_sales
On daily_activity_concede.pid = open_sales.Project_No
inner join matrix_sow
on daily_activity_concede.SOW = matrix_sow.sow_code
inner join employee_details
on daily_activity_concede.NIK_ENG = employee_details.nik or daily_activity_concede.NIK_Tech_1 = employee_details.nik
or daily_activity_concede.NIK_Tech_2 = employee_details.nik or daily_activity_concede.NIK_Tech_3 = employee_details.nik
or daily_activity_concede.NIK_Tech_4 = employee_details.nik
Where daily_activity_concede.PID = "SE.17.03.08.0221"
